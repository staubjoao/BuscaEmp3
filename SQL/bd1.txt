CREATE TABLE estado (
                idestado INTEGER NOT NULL AUTO_INCREMENT,
                estado VARCHAR(45) NOT NULL,
                uf CHAR(2) NOT NULL,
                CONSTRAINT estado_pk PRIMARY KEY (idestado)
);


CREATE TABLE cidade (
                idcidade INTEGER NOT NULL AUTO_INCREMENT,
                idestado INTEGER NOT NULL,
                cidade VARCHAR(45) NOT NULL,
                CONSTRAINT cidade_pk PRIMARY KEY (idcidade, idestado)
);


CREATE TABLE empresa (
                idempresa INTEGER NOT NULL AUTO_INCREMENT,
                idcidade INTEGER NOT NULL,
                idestado INTEGER NOT NULL,
                email VARCHAR(80) NOT NULL,
                senha VARCHAR(12) NOT NULL,
                endereco VARCHAR(80) NOT NULL,
                numero VARCHAR(45) NOT NULL,
                cep CHAR(8) NOT NULL,
                nomeEmpresa VARCHAR(80) NOT NULL,
                cnpj CHAR(14) NOT NULL,
                ramo VARCHAR(80) NOT NULL,
                telefone VARCHAR(45) NOT NULL,
                CONSTRAINT empresa_pk PRIMARY KEY (idempresa, idcidade, idestado)
);
